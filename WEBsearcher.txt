      <h1>Zoek bedrijfsactiviteiten + logo</h1>

      <form id="companyForm">
        <input id="companyInput" type="text" placeholder="Typ een bedrijfsnaam (bv. 'Philips', 'ASML', 'IKEA')..." required />
        <button id="submitBtn" type="submit">Zoeken</button>
      </form>

      <div class="meta">Bronnen: Wikipedia (samenvatting) + Wikidata (website/logo). Logo fallback: Clearbit op basis van domein.</div>

      <div id="status" class="status"></div>

      <div id="result" class="result" style="display:none;">
        <div class="logo" id="logoBox">
          <span>Geen logo</span>
        </div>

        <div>
          <p class="kicker" id="kicker"></p>
          <h2 class="title" id="title"></h2>
          <p class="summary" id="summary"></p>

          <div class="links" id="links"></div>

          <div class="small">
            Tip: als je een heel nieuw/niche bedrijf zoekt, kan het zijn dat er (nog) geen Wikipedia/Wikidata-item is.

  <script>
    const $ = (sel) => document.querySelector(sel);

    const form = $("#companyForm");
    const input = $("#companyInput");
    const statusEl = $("#status");
    const resultEl = $("#result");
    const submitBtn = $("#submitBtn");

    const logoBox = $("#logoBox");
    const kickerEl = $("#kicker");
    const titleEl = $("#title");
    const summaryEl = $("#summary");
    const linksEl = $("#links");

    function setStatus(msg, isError=false) {
      statusEl.textContent = msg || "";
      statusEl.className = "status" + (isError ? " error" : "");
    }

    function resetUI() {
      resultEl.style.display = "none";
      logoBox.innerHTML = "<span>Geen logo</span>";
      kickerEl.textContent = "";
      titleEl.textContent = "";
      summaryEl.textContent = "";
      linksEl.innerHTML = "";
    }

    function safeBadge(href, label) {
      const a = document.createElement("a");
      a.className = "badge";
      a.href = href;
      a.target = "_blank";
      a.rel = "noopener noreferrer";
      a.textContent = label;
      return a;
    }

    async function wikiSearch(companyName) {
      // Zoek beste match in NL, fallback EN
      async function searchOn(lang) {
        const url = `https://${lang}.wikipedia.org/w/api.php?action=query&list=search&srsearch=${encodeURIComponent(companyName)}&format=json&origin=*`;
        const res = await fetch(url);
        if (!res.ok) throw new Error("Wikipedia search mislukt");
        const data = await res.json();
        return { lang, results: data?.query?.search || [] };
      }

      let nl = await searchOn("nl");
      if (nl.results.length) return { lang: "nl", title: nl.results[0].title };

      let en = await searchOn("en");
      if (en.results.length) return { lang: "en", title: en.results[0].title };

      return null;
    }

    async function wikiSummary(lang, title) {
      const url = `https://${lang}.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(title)}`;
      const res = await fetch(url);
      if (!res.ok) throw new Error("Wikipedia summary mislukt");
      const data = await res.json();
      return {
        title: data?.title || title,
        extract: data?.extract || "",
        pageUrl: data?.content_urls?.desktop?.page || null
      };
    }

    async function wikidataCompanyInfo(companyName) {
      // 1) Zoek entity via mwapi en pak P856 (website) en P154 (logo) indien beschikbaar.
      // We proberen ook een beetje te filteren op "bedrijf/organisatie" via instance of (P31) waar mogelijk.
      const query = `
SELECT ?item ?itemLabel ?website ?logo WHERE {
  SERVICE wikibase:mwapi {
    bd:serviceParam wikibase:endpoint "www.wikidata.org";
    bd:serviceParam wikibase:api "EntitySearch";
    bd:serviceParam mwapi:search "${companyName.replace(/"/g, '\\"')}";
    bd:serviceParam mwapi:language "nl";
    ?item wikibase:apiOutputItem mwapi:item.
  }
  OPTIONAL { ?item wdt:P856 ?website. }
  OPTIONAL { ?item wdt:P154 ?logo. }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "nl,en". }
}
LIMIT 5
      `.trim();

      const url = "https://query.wikidata.org/sparql?format=json&query=" + encodeURIComponent(query);
      const res = await fetch(url, { headers: { "Accept": "application/sparql-results+json" } });
      if (!res.ok) throw new Error("Wikidata query mislukt");

      const data = await res.json();
      const rows = data?.results?.bindings || [];

      if (!rows.length) return null;

      // kies eerste rij die minstens website of logo heeft, anders de eerste
      const pick = rows.find(r => r.website?.value || r.logo?.value) || rows[0];

      return {
        label: pick.itemLabel?.value || null,
        website: pick.website?.value || null,
        logoFile: pick.logo?.value || null, // meestal "File:....svg"
        itemUrl: pick.item?.value || null
      };
    }

    function domainFromUrl(url) {
      try {
        const u = new URL(url);
        return u.hostname;
      } catch {
        return null;
      }
    }

    function commonsFileUrl(fileTitle) {
      // fileTitle meestal "http://www.wikidata.org/entity/..."? Nee, wdt:P154 geeft vaak string "File:..."
      // Voor Wikimedia Commons bestandslink: Special:FilePath/<bestandsnaam>
      // We strippen "File:" en encoden de rest.
      if (!fileTitle) return null;
      const cleaned = fileTitle.replace(/^File:/i, "");
      return "https://commons.wikimedia.org/wiki/Special:FilePath/" + encodeURIComponent(cleaned);
    }

    function setLogoFromUrl(imgUrl) {
      logoBox.innerHTML = "";
      const img = document.createElement("img");
      img.alt = "Logo";
      img.loading = "lazy";
      img.referrerPolicy = "no-referrer";
      img.src = imgUrl;

      img.onerror = () => {
        // Als logo niet laadt, laat placeholder staan
        logoBox.innerHTML = "<span>Geen logo</span>";
      };

      logoBox.appendChild(img);
    }

    form.addEventListener("submit", async (e) => {
      e.preventDefault();
      const companyName = input.value.trim();
      if (!companyName) return;

      resetUI();
      submitBtn.disabled = true;
      setStatus("Bezig met zoeken…");

      try {
        // Parallel: Wikidata + Wikipedia
        const [wd, ws] = await Promise.all([
          wikidataCompanyInfo(companyName).catch(() => null),
          wikiSearch(companyName).catch(() => null)
        ]);

        // Wikidata: logo/website
        if (wd?.logoFile) {
          const commonsUrl = commonsFileUrl(wd.logoFile);
          if (commonsUrl) setLogoFromUrl(commonsUrl);
        } else if (wd?.website) {
          const dom = domainFromUrl(wd.website);
          if (dom) setLogoFromUrl(`https://logo.clearbit.com/${encodeURIComponent(dom)}`);
        }

        // Wikipedia: summary
        let summaryData = null;
        if (ws?.lang && ws?.title) {
          summaryData = await wikiSummary(ws.lang, ws.title).catch(() => null);
        }

        // UI vullen
        const displayTitle = summaryData?.title || wd?.label || companyName;
        titleEl.textContent = displayTitle;

        kickerEl.textContent =
          (summaryData ? `Wikipedia (${ws.lang.toUpperCase()})` : "Geen Wikipedia-samenvatting gevonden")
          + (wd?.itemUrl ? " • Wikidata" : "");

        summaryEl.textContent =
          summaryData?.extract
          || "Geen samenvatting gevonden. Probeer een andere schrijfwijze of een meer specifieke naam (bijv. met land/branche).";

        // Links
        linksEl.innerHTML = "";
        if (summaryData?.pageUrl) linksEl.appendChild(safeBadge(summaryData.pageUrl, "Wikipedia-pagina"));
        if (wd?.itemUrl) linksEl.appendChild(safeBadge(wd.itemUrl, "Wikidata-item"));
        if (wd?.website) linksEl.appendChild(safeBadge(wd.website, "Officiële website"));

        resultEl.style.display = "grid";
        setStatus("");
      } catch (err) {
        console.error(err);
        setStatus("Er ging iets mis bij het ophalen van data. Probeer opnieuw.", true);
      } finally {
        submitBtn.disabled = false;
      }
    });
  </script>
